def initialize(fetcher, cache, debug, poll_interval_millis: FLAG_CONFIG_POLLING_INTERVAL_MILLIS)
  @fetcher = fetcher
  @cache = cache
  @poll_interval_millis = poll_interval_millis
  @logger = Logger.new($stdout)
  @debug = debug
  @poller_thread = nil
  @is_running = false
end
